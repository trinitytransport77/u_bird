require("dotenv").config();
const express = require("express");
const mongoose = require("mongoose");
const cors = require("cors");
const jwt = require("jsonwebtoken");
const WebSocket = require("ws");

const app = express();
app.use(cors());
app.use(express.json());

// MongoDB Connection
mongoose
  .connect(process.env.MONGO_URL)
  .then(() => console.log("MongoDB Connected"))
  .catch((err) => console.log(err));

// WebSocket Server
const wss = new WebSocket.Server({ noServer: true });

let drivers = {}; // store driver locations
let riders = {};  // store rider requests

wss.on("connection", (ws) => {
  ws.on("message", (msg) => {
    const data = JSON.parse(msg);

    // Driver location updates
    if (data.type === "driver_location") {
      drivers[data.driverId] = data.location;
      broadcast({ type: "drivers_update", drivers });
    }

    // Rider requests a ride
    if (data.type === "ride_request") {
      riders[data.riderId] = data;
      broadcast({ type: "new_ride_request", request: data });
    }
  });
});

// Broadcast to all clients
function broadcast(data) {
  wss.clients.forEach((client) => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(JSON.stringify(data));
    }
  });
}

// Express Routes
app.get("/", (req, res) => {
  res.send("U_BIRD Backend Running");
});

// Start Server
const server = app.listen(process.env.PORT || 5000, () =>
  console.log("Server running")
);

// Upgrade HTTP → WebSocket
server.on("upgrade", (req, socket, head) => {
  wss.handleUpgrade(req, socket, head, (ws) => {
    wss.emit("connection", ws, req);
  });
});
